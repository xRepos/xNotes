Window Resize

To resize any part of the page upon window resize:

function windowResize() {
    $('#map').height($(window).height() - 50);
}
$(window).resize(windowResize);

Or

function windowResize() {
    $('#map').css('height', function() {return $(window).height() - 50});
}

$(window).resize(windowResize);
