Loading Data in a Service

angular.module('myModule', []) 
    .factory('myService', ['$http', function($http) {
        var item = {};
        $http.get(...).success(function(data) {
            item.foo = data;
        });
        return item;
    }])

function myController($scope, myService) {
    $scope.item = myService.getItem();
}
